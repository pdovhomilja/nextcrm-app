### Step 1: Create Security Analytics Service (30 minutes)

**File:** `lib/security/security-analytics.ts` (new file)

**🚀 SECURITY-FIRST WRAPPER APPROACH**: Use existing audit data instead of creating new logging

```typescript
import { withCompanyAccessValidation } from "@/lib/security/company-access-validator";
import { auth } from "@/auth";
import db from "@/lib/db";

export interface SecurityMetrics {
  totalEvents: number;
  riskDistribution: {
    critical: number;
    high: number;
    medium: number;
    low: number;
  };
  recentEvents: SecurityEvent[];
  topViolators: {
    userId: string;
    email: string;
    violationCount: number;
  }[];
  resourceTypeBreakdown: {
    task: number;
    board: number;
    ai_query: number;
    other: number;
  };
}

export interface SecurityEvent {
  id: string;
  userId: string;
  email?: string;
  action: string;
  resource: string;
  risk: string;
  timestamp: Date;
  details: any;
}

export class SecurityAnalyticsService {
  
  /**
   * Get security metrics for dashboard with company access validation
   */
  static async getSecurityMetrics(
    companyId: string, 
    timeRange: "24h" | "7d" | "30d" = "24h"
  ): Promise<SecurityMetrics> {
    const session = await auth();
    if (!session?.user?.id) {
      throw new Error("Unauthorized");
    }

    // 🔒 SECURITY-FIRST WRAPPER: Automatic company validation + audit logging
    return withCompanyAccessValidation(
      session.user.id,
      companyId,
      "board",  // Resource type for security dashboard access
      "security_metrics", // Action
      async () => {
        const since = new Date();
        switch (timeRange) {
          case "24h":
            since.setHours(since.getHours() - 24);
            break;
          case "7d":
            since.setDate(since.getDate() - 7);
            break;
          case "30d":
            since.setDate(since.getDate() - 30);
            break;
        }

        // ✅ SIMPLIFIED QUERY - Security handled by wrapper
        const events = await db.securityAuditLog.findMany({
          where: {
            timestamp: { gte: since },
            // Query all security events for this company context
            OR: [
              {
                details: {
                  path: ["companyId"],
                  equals: companyId,
                },
              },
              {
                // Also include events where user accessed company resources
                AND: [
                  {
                    details: {
                      path: ["companyId"],
                      not: null,
                    },
                  },
                  {
                    details: {
                      path: ["companyId"],
                      equals: companyId,
                    },
                  },
                ],
              },
            ],
          },
          include: {
            user: {
              select: {
                id: true,
                email: true,
              },
            },
          },
          orderBy: { timestamp: "desc" },
          take: 200,
        });

        const riskDistribution = {
          critical: events.filter(e => e.risk === "high" && e.authorized === false).length,
          high: events.filter(e => e.risk === "high" && e.authorized === true).length,
          medium: events.filter(e => e.risk === "low" && e.authorized === false).length,
          low: events.filter(e => e.risk === "low" && e.authorized === true).length,
        };

        const resourceTypeBreakdown = {
          task: events.filter(e => e.resource === "task").length,
          board: events.filter(e => e.resource === "board").length,
          ai_query: events.filter(e => e.resource === "ai_query").length,
          other: events.filter(e => !["task", "board", "ai_query"].includes(e.resource)).length,
        };

        // Find users with most violations (high risk + not authorized)
        const violationCounts = new Map<string, { email: string; count: number }>();
        events
          .filter(e => e.risk === "high" && !e.authorized)
          .forEach(event => {
            const existing = violationCounts.get(event.userId) || { 
              email: event.user?.email || "Unknown", 
              count: 0 
            };
            violationCounts.set(event.userId, { 
              ...existing, 
              count: existing.count + 1 
            });
          });

        const topViolators = Array.from(violationCounts.entries())
          .map(([userId, data]) => ({
            userId,
            email: data.email,
            violationCount: data.count,
          }))
          .sort((a, b) => b.violationCount - a.violationCount)
          .slice(0, 5);

        const recentEvents: SecurityEvent[] = events.slice(0, 50).map(event => ({
          id: event.id,
          userId: event.userId,
          email: event.user?.email,
          action: event.action,
          resource: event.resource,
          risk: event.authorized ? event.risk : "violation",
          timestamp: event.timestamp,
          details: event.details,
        }));

        return {
          totalEvents: events.length,
          riskDistribution,
          recentEvents,
          topViolators,
          resourceTypeBreakdown,
        };
      }
    );
  }

  /**
   * Check if user has suspicious activity patterns
   */
  static async checkSuspiciousActivity(
    userId: string, 
    companyId: string
  ): Promise<{
    isSuspicious: boolean;
    riskScore: number;
    reasons: string[];
  }> {
    const session = await auth();
    if (!session?.user?.id) {
      throw new Error("Unauthorized");
    }

    // 🔒 SECURITY-FIRST WRAPPER: Validate access to check suspicious activity
    const result = await withCompanyAccessValidation(
      session.user.id,
      companyId,
      "board",
      "suspicious_activity_check",
      async () => {
        const since = new Date();
        since.setHours(since.getHours() - 1); // Check last hour

        const recentEvents = await db.securityAuditLog.findMany({
          where: {
            userId,
            timestamp: { gte: since },
            risk: "high",
            authorized: false, // Only violations
          },
        });

        const riskScore = recentEvents.length * 10;
        const reasons: string[] = [];

        if (recentEvents.length >= 5) {
          reasons.push("Multiple security violations in short time");
        }

        const unauthorizedAttempts = recentEvents.filter(e => 
          e.action.includes("access") && !e.authorized
        );
        if (unauthorizedAttempts.length >= 3) {
          reasons.push("Multiple unauthorized access attempts");
        }

        const aiQueries = recentEvents.filter(e => 
          e.resource === "ai_query" && !e.authorized
        );
        if (aiQueries.length >= 2) {
          reasons.push("Unauthorized AI tool access attempts");
        }

        return {
          isSuspicious: riskScore >= 30 || reasons.length >= 2,
          riskScore,
          reasons,
        };
      }
    );

    if (!result.success) {
      throw new Error(result.error || "Failed to check suspicious activity");
    }

    return result.data!;
  }
}
```

```typescript
  AlertTriangle, 
  Shield, 
  Eye, 
  Users, 
  Activity,
  Database,
  Bot,
  FileText 
} from "lucide-react";
import { SecurityMetrics, SecurityEvent } from "@/lib/security/security-analytics";

interface SecurityDashboardProps {
  companyId: string;
}

export function SecurityDashboard({ companyId }: SecurityDashboardProps) {
  const [metrics, setMetrics] = useState<SecurityMetrics | null>(null);
  const [loading, setLoading] = useState(true);
  const [selectedTimeRange, setSelectedTimeRange] = useState<"24h" | "7d" | "30d">("24h");

  useEffect(() => {
    const fetchMetrics = async () => {
      try {
        const response = await fetch(
          `/api/security/analytics?companyId=${companyId}&timeRange=${selectedTimeRange}`
        );
        const data = await response.json();
        if (data.success) {
          setMetrics(data.data);
        }
      } catch (error) {
        console.error("Failed to fetch security metrics:", error);
      } finally {
        setLoading(false);
      }
    };

    fetchMetrics();
    // Refresh every 30 seconds
    const interval = setInterval(fetchMetrics, 30000);
    return () => clearInterval(interval);
  }, [companyId, selectedTimeRange]);

  if (loading) {
    return <div>Loading security analytics...</div>;
  }

  if (!metrics) {
    return <div>Failed to load security analytics</div>;
  }

  const highRiskEvents = metrics.riskDistribution.critical + metrics.riskDistribution.high;
  const totalResourceAccess = Object.values(metrics.resourceTypeBreakdown).reduce((a, b) => a + b, 0);

  return (
    <div className="space-y-6">
      {/* Time Range Selector */}
      <div className="flex space-x-2">
        {(["24h", "7d", "30d"] as const).map((range) => (
          <button
            key={range}
            onClick={() => setSelectedTimeRange(range)}
            className={`px-3 py-1 rounded text-sm ${
              selectedTimeRange === range 
                ? "bg-primary text-primary-foreground" 
                : "bg-secondary"
            }`}
          >
            {range === "24h" ? "24 Hours" : range === "7d" ? "7 Days" : "30 Days"}

          <AlertTriangle className="h-4 w-4" />
          <AlertDescription>
            {highRiskEvents} security violations detected in the selected time range. 
            Review the audit trail below.
          </AlertDescription>
        </Alert>
      )}

      {/* Security Overview Cards */}
      <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-4">
        <Card>
          <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
            <CardTitle className="text-sm font-medium">Total Events</CardTitle>
            <Eye className="h-4 w-4 text-muted-foreground" />
          </CardHeader>
          <CardContent>
            <div className="text-2xl font-bold">{metrics.totalEvents}</div>
            <p className="text-xs text-muted-foreground">All security events</p>
          </CardContent>
        </Card>

        <Card>

            <CardTitle className="text-sm font-medium">Violations</CardTitle>
            <AlertTriangle className="h-4 w-4 text-red-600" />

                <p className="text-2xl font-bold">{metrics.resourceTypeBreakdown.task}</p>
              </div>
            </div>

              <Database className="h-4 w-4 text-purple-500" />
```

### Step 3: Create Security Analytics API Endpoint (30 minutes)

**File:** `app/api/security/analytics/route.ts` (new file)

**🚀 SECURITY-FIRST WRAPPER APPROACH**: API endpoint uses existing validator

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/auth";
import { SecurityAnalyticsService } from "@/lib/security/security-analytics";

export async function GET(request: NextRequest) {
  try {
    const session = await auth();
    if (!session?.user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const url = new URL(request.url);
    const companyId = url.searchParams.get("companyId");
    const timeRange = url.searchParams.get("timeRange") as "24h" | "7d" | "30d" || "24h";

    if (!companyId) {
      return NextResponse.json(
        { error: "Company ID required" },
        { status: 400 }
      );
    }

    // 🔒 SECURITY-FIRST WRAPPER: Service method handles validation + audit logging
    const result = await SecurityAnalyticsService.getSecurityMetrics(companyId, timeRange);

    if (!result.success) {
      return NextResponse.json(
        { error: result.error },
        { status: 403 }
      );
    }

    return NextResponse.json(result);
  } catch (error) {
    console.error("Security analytics API error:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}
```
